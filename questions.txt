- How many times has each FF# been used?

SELECT p.frequent_flyer_number, count(p.id)
FROM passengers p
WHERE p.frequent_flyer_number IS NOT NULL
GROUP BY p.frequent_flyer_number
ORDER BY count(p.id) DESC

    count(p.id)
